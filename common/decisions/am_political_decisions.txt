monarchism_on_the_rise = {

	open_up_political_discourse_monarchism = {

		icon = generic_political_discourse

		available = {
			monarchism < 0.9
			democratic < 0.9
			communism < 0.9
			neutrality < 0.9
			fascism < 0.9 
			theocratic < 0.9
		}

		visible = {
			has_idea_with_trait = royal_supporter
			NOT = { has_country_flag = open_up_political_discourse_monarchism }
			#check_has_focus_tree_to_switch_to_monarchism = yes
			NOT = { has_government = monarchism }
		}
		
		remove_trigger = {
			has_government = monarchism
		}
		
		fire_only_once = yes
		
		cost = 75
		
		complete_effect = {
			set_country_flag = open_up_political_discourse_monarchism
		}
		
		ai_will_do = {
			factor = 1
		}
	}
	
		expand_civil_monarchism_support = {

		icon = generic_civil_support

		visible = {
			has_country_flag = open_up_political_discourse_monarchism
			has_idea_with_trait = royal_supporter
			#check_has_focus_tree_to_switch_to_fascism = yes
			NOT = { has_government = monarchism }
		}

		cancel_trigger = {
			hidden_trigger = {
				has_civil_war = yes
			}
		}
		
		fire_only_once = no
		
		cost = 25

		days_remove = 30

		modifier = {
			monarchism_drift = 0.02
			stability_weekly = -0.03
 			political_power_gain = -0.2
		}
		
		ai_will_do = {
			factor = 1
		}
	}
	
	
	discredit_government_monarchism = {

		icon = generic_political_discourse

		available = {
			monarchism < 0.8
			democratic < 0.8
			communism < 0.8
			neutrality < 0.8
			fascism < 0.8 
			theocratic < 0.8
			has_idea_with_trait = royal_supporter
		}

		visible = {
			has_country_flag = open_up_political_discourse_monarchism
			#check_has_focus_tree_to_switch_to_monarchism = yes
			NOT = { has_government = monarchism }
		}
		
		remove_trigger = {
			has_government = monarchism
		}
		
		fire_only_once = yes
		
		cost = 50
		
		complete_effect = {
			add_stability = -0.010
			if = {
				limit = {
					has_government = democratic
				}
				add_popularity = {
					ideology = democratic
					popularity = -0.1
				}
			}
			if = {
				limit = {
					has_government = theocratic
				}
				add_popularity = {
					ideology = theocratic
					popularity = -0.1
				}
			}
			if = {
				limit = {
					has_government = communism
				}
				add_popularity = {
					ideology = communism
					popularity = -0.1
				}
			}
			if = {
				limit = {
					has_government = neutrality
				}
				add_popularity = {
					ideology = neutrality
					popularity = -0.1
				}
			}
			if = {
				limit = {
					has_government = fascism
				}
				add_popularity = {
					ideology = fascism
					popularity = -0.1
				}
			}
		}
		
		ai_will_do = {
			factor = 1
		}
	}
	
	hold_the_monarchist_national_referendum = {

		icon = generic_political_discourse

		available = {
			monarchism > 0.5
		}

		visible = {
			NOT = { has_government = monarchism }
			#check_has_focus_tree_to_switch_to_monarchism = yes
			OR = {
				monarchism > 0.5
				has_country_flag = open_up_political_discourse_monarchism
			}
		}
		
		remove_trigger = {
			has_government = monarchism
		}
		
		fire_only_once = yes
		
		cost = 100
		
		complete_effect = {
			country_event = { id = newpolitical.1 }
		}
		
		ai_will_do = {
			factor = 210
			modifier = {
				monarchism > 0.6
				factor = 10
			}
		}
	}
}
